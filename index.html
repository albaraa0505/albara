// ميزانية الشاليه — Apps Script v2 // دعم الدفع عن أشهر سابقة const SHEET_ID = '1KeNM2z3s1m_6aN5_Qg-gGQd3WqN5wygx3csb0BTR700'; const MEMBERS = ['عايشة','البراء','فاطمة','بلقيس','عبدالله']; const MEMBER_FUNDS = { 'عايشة': { طوارئ: 12, مصاريف: 0 }, 'البراء': { طوارئ: 0, مصاريف: 12 }, 'فاطمة': { طوارئ: 6, مصاريف: 6 }, 'بلقيس': { طوارئ: 0, مصاريف: 12 }, 'عبدالله':{ طوارئ: 0, مصاريف: 12 } }; const MONTHS_AR = ['يناير','فبراير','مارس','أبريل','مايو','يونيو', 'يوليو','أغسطس','سبتمبر','أكتوبر','نوفمبر','ديسمبر']; function getMonthKey(date) { return MONTHS_AR[date.getMonth()] + ' ' + date.getFullYear(); } function getCurrentMonth() { return getMonthKey(new Date()); } // كل الأشهر من بداية 2025 حتى الشهر الحالي function getAllMonths() { const months = []; const now = new Date(); const start = new Date(2025, 0, 1); let d = new Date(start); while (d <= now) { months.push(getMonthKey(d)); d.setMonth(d.getMonth() + 1); } return months.reverse(); // الأحدث أولاً } function initSheets() { const ss = SpreadsheetApp.openById(SHEET_ID); let ws = ss.getSheetByName('الدفعات'); if (!ws) { ws = ss.insertSheet('الدفعات'); ws.appendRow(['الشهر','الاسم','المبلغ للمصاريف','المبلغ للطوارئ','التاريخ والوقت']); ws.getRange(1,1,1,5).setFontWeight('bold').setBackground('#8B6F47').setFontColor('#FFFFFF'); ws.setColumnWidth(1,120); ws.setColumnWidth(2,100); ws.setColumnWidth(3,130); ws.setColumnWidth(4,130); ws.setColumnWidth(5,160); } let ws2 = ss.getSheetByName('الإعدادات'); if (!ws2) { ws2 = ss.insertSheet('الإعدادات'); ws2.appendRow(['البيان','القيمة']); ws2.appendRow(['رصيد افتتاحي مصاريف', 0]); ws2.appendRow(['رصيد افتتاحي طوارئ', 0]); ws2.appendRow(['مصاريف ثابتة شهرية', 42]); ws2.getRange(1,1,1,2).setFontWeight('bold').setBackground('#8B6F47').setFontColor('#FFFFFF'); } let ws3 = ss.getSheetByName('السحوبات'); if (!ws3) { ws3 = ss.insertSheet('السحوبات'); ws3.appendRow(['التاريخ','السبب','المبلغ','الصندوق','ملاحظات']); ws3.getRange(1,1,1,5).setFontWeight('bold').setBackground('#8B6F47').setFontColor('#FFFFFF'); } } function getData() { try { initSheets(); const ss = SpreadsheetApp.openById(SHEET_ID); const currentMonth = getCurrentMonth(); const allMonths = getAllMonths(); // كل الدفعات const ws = ss.getSheetByName('الدفعات'); const rows = ws.getDataRange().getValues(); // بناء خريطة: شهر → عضو → بيانات const allPayments = {}; allMonths.forEach(m => { allPayments[m] = {}; MEMBERS.forEach(mem => allPayments[m][mem] = null); }); let totalExp = 0, totalEm = 0; for (let i = 1; i < rows.length; i++) { const row = rows[i]; const month = row[0]; const name = row[1]; if (allPayments[month] !== undefined && MEMBERS.includes(name)) { allPayments[month][name] = { مصاريف: row[2], طوارئ: row[3], date: row[4] }; totalExp += Number(row[2]) || 0; totalEm += Number(row[3]) || 0; } } // الإعدادات const ws2 = ss.getSheetByName('الإعدادات'); const settings = ws2.getDataRange().getValues(); let openingExp = 0, openingEm = 0; for (let i = 1; i < settings.length; i++) { if (settings[i][0] === 'رصيد افتتاحي مصاريف') openingExp = Number(settings[i][1]) || 0; if (settings[i][0] === 'رصيد افتتاحي طوارئ') openingEm = Number(settings[i][1]) || 0; } totalExp += openingExp; totalEm += openingEm; // السحوبات const ws3 = ss.getSheetByName('السحوبات'); const withdrawals = ws3.getDataRange().getValues(); for (let i = 1; i < withdrawals.length; i++) { const amt = Number(withdrawals[i][2]) || 0; if (withdrawals[i][3] === 'مصاريف') totalExp -= amt; if (withdrawals[i][3] === 'طوارئ') totalEm -= amt; } // الأرشيف: كل الأشهر ماعدا الحالي const archive = {}; allMonths.filter(m => m !== currentMonth).forEach(m => { archive[m] = allPayments[m]; }); return JSON.stringify({ ok: true, currentMonth, allMonths, payments: allPayments[currentMonth], balanceExp: totalExp, balanceEm: totalEm, archive, memberFunds: MEMBER_FUNDS }); } catch(e) { return JSON.stringify({ ok: false, error: e.message }); } } function recordPayment(name, month) { try { if (!MEMBERS.includes(name)) return JSON.stringify({ ok: false, error: 'اسم غير صحيح' }); const ss = SpreadsheetApp.openById(SHEET_ID); const ws = ss.getSheetByName('الدفعات'); const targetMonth = month || getCurrentMonth(); // تحقق: هل دفع هذا الشهر مسبقاً؟ const rows = ws.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (rows[i][0] === targetMonth && rows[i][1] === name) { return JSON.stringify({ ok: false, error: `${name} دفع ${targetMonth} مسبقاً ✅` }); } } const funds = MEMBER_FUNDS[name]; const dateStr = Utilities.formatDate(new Date(), 'Asia/Riyadh', 'dd/MM/yyyy HH:mm'); ws.appendRow([targetMonth, name, funds.مصاريف, funds.طوارئ, dateStr]); const lastRow = ws.getLastRow(); ws.getRange(lastRow,1,1,5).setBackground('#E8F5E9'); return JSON.stringify({ ok: true, message: `✅ تم تسجيل دفعة ${name} — ${targetMonth}` }); } catch(e) { return JSON.stringify({ ok: false, error: e.message }); } } function doGet(e) { const action = e.parameter.action || 'data'; const name = decodeURIComponent(e.parameter.name || ''); const month = decodeURIComponent(e.parameter.month || ''); let result; if (action === 'pay') { result = recordPayment(name, month || null); } else { result = getData(); } return ContentService .createTextOutput(result) .setMimeType(ContentService.MimeType.JSON); } function test() { Logger.log(getData()); }
